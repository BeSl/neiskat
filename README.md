### получить ссылку на файл в элементе
```bsl
ФайлСсылка.СоздатьHttpСсылку().Абсолютная()
```

### авторизоваться в сервисе элемента из 1С
требуется общий модуль HTTPКоннектор
```bsl
  Client_Id = "";
  Client_Secret = ""; //получаем при создании пользоватля 1 раз
	АдресАвторизации = "sys/token";
	
	АдресСервиса = "https://[АДРЕС].1cmycloud.com/applications/[АДРЕС2]";
	ЗаголовкиЗапроса = Новый Соответствие;
	ЗаголовкиЗапроса.Вставить("Content-Type", "application/x-www-form-urlencoded");
	
	Аутентификация = Новый Структура;
	Аутентификация.Вставить("Пользователь", Client_Id);
	Аутентификация.Вставить("Пароль", Client_Secret);
	
	ПараметрыЗапроса = Новый Структура;
	ПараметрыЗапроса.Вставить("Заголовки", ЗаголовкиЗапроса);
	ПараметрыЗапроса.Вставить("Аутентификация", Аутентификация);
	урл = СтрШаблон("%1/%2", АдресСервиса, АдресАвторизации);
	ответ = HTTPКоннектор.Post_v_2_1_3(урл, "grant_type=client_credentials", ПараметрыЗапроса);
	Если ответ.КодСостояния <> 200 Тогда
		ОписаниеОшибки_ = ОписаниеОшибки_ + " Ошибка авторизации";
	КонецЕсли;
	
	ответСтруктурой = HTTPКоннектор.КакJson(ответ);
	ТокенСессии = ответСтруктурой.Получить("id_token");	
```

### отправка данных, с полученным токеном сессии в сервисе элемента из 1С
Примечание: токен действителен 1 час
```bsl
  строкаОтправки =	"МОЯ ВАЖНАЯ СТРОКА";
	ЗаголовкиЗапроса = Новый Соответствие;
	ЗаголовкиЗапроса.Вставить("Content-Type", "application/json");
	ЗаголовкиЗапроса.Вставить("Authorization", "Bearer "+ТокенСессии);
	АдресХэндлера = "api/[НАШ_СЕРВИС]";
	ПараметрыЗапроса = Новый Структура;
	ПараметрыЗапроса.Вставить("Заголовки", ЗаголовкиЗапроса);
	
	ВесьURL = СтрШаблон("%1/%2", АдресСервиса, АдресХэндлера);
	ответ = КоннекторHTTP.Post(ВесьURL, строкаОтправки, ПараметрыЗапроса);
```
